/**
 * ============================================================
 *  CAN1 初始化参考代码 (STM32F407VET6, PD0/PD1)
 *  波特率: 1 Mbps
 *  APB1 = 42 MHz, Prescaler = 3, BS1 = 11TQ, BS2 = 2TQ
 *  => 42MHz / (3 * (1+11+2)) = 42MHz / 42 = 1 Mbps
 * ============================================================
 *
 *  如果你用 CubeMX 生成的工程，只需要在 CubeMX 里配置：
 *    - CAN1, Activated
 *    - Prescaler = 3
 *    - Time Quanta in Bit Segment 1 = 11 Times
 *    - Time Quanta in Bit Segment 2 = 2 Times
 *    - ReSynchronization Jump Width = 1 Time
 *    - Operating Mode = Normal
 *    - GPIO: PD0 = CAN1_RX, PD1 = CAN1_TX
 *    - NVIC: CAN1 RX0 interrupt = Enabled
 *
 *  如果手写初始化，参考以下代码（放在你工程的 can.c 中）：
 */

#include "stm32f4xx_hal.h"

CAN_HandleTypeDef hcan1;

void MX_CAN1_Init(void)
{
    hcan1.Instance = CAN1;
    hcan1.Init.Prescaler = 3;
    hcan1.Init.Mode = CAN_MODE_NORMAL;
    hcan1.Init.SyncJumpWidth = CAN_SJW_1TQ;
    hcan1.Init.TimeSeg1 = CAN_BS1_11TQ;
    hcan1.Init.TimeSeg2 = CAN_BS2_2TQ;
    hcan1.Init.TimeTriggeredMode = DISABLE;
    hcan1.Init.AutoBusOff = ENABLE;
    hcan1.Init.AutoWakeUp = DISABLE;
    hcan1.Init.AutoRetransmission = ENABLE;
    hcan1.Init.ReceiveFifoLocked = DISABLE;
    hcan1.Init.TransmitFifoPriority = DISABLE;

    if (HAL_CAN_Init(&hcan1) != HAL_OK)
    {
        Error_Handler();
    }
}

/* GPIO 和时钟配置 (HAL_CAN_MspInit 回调) */
void HAL_CAN_MspInit(CAN_HandleTypeDef *canHandle)
{
    GPIO_InitTypeDef GPIO_InitStruct = {0};

    if (canHandle->Instance == CAN1)
    {
        __HAL_RCC_CAN1_CLK_ENABLE();
        __HAL_RCC_GPIOD_CLK_ENABLE();

        /* PD0 = CAN1_RX, PD1 = CAN1_TX */
        GPIO_InitStruct.Pin = GPIO_PIN_0 | GPIO_PIN_1;
        GPIO_InitStruct.Mode = GPIO_MODE_AF_PP;
        GPIO_InitStruct.Pull = GPIO_NOPULL;
        GPIO_InitStruct.Speed = GPIO_SPEED_FREQ_VERY_HIGH;
        GPIO_InitStruct.Alternate = GPIO_AF9_CAN1;
        HAL_GPIO_Init(GPIOD, &GPIO_InitStruct);

        /* CAN1 RX0 中断 */
        HAL_NVIC_SetPriority(CAN1_RX0_IRQn, 1, 0);
        HAL_NVIC_EnableIRQ(CAN1_RX0_IRQn);
    }
}
